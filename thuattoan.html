2. Binary Search

int binarySearch(int a[], int n, int x){
    int l=0, r=n-1;
    while(l<=r){
        int mid=(l+r)/2;
        if(a[mid]==x) return mid;
        if(a[mid]<x) l=mid+1;
        else r=mid-1;
    }
    return -1;
}

def binary_search(a, x):
    l, r = 0, len(a)-1
    while l <= r:
        mid = (l+r)//2
        if a[mid] == x:
            return mid
        elif a[mid] < x:
            l = mid+1
        else:
            r = mid-1
    return -1
